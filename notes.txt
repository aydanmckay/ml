to obtain rave catalaogue:

SELECT t1.rave_obs_id, t1.ra_input, t1.dec_input, t2.teff_cal_madera, t2.teff_error_madera, t2.logg_cal_madera, t2.logg_error_madera, t2.m_h_cal_madera, t2.m_h_error_madera, t2.snr_madera, t2.algo_conv_madera,
t3.spectrum_png, t3.spectrum_fits
FROM ravedr6.dr6_obsdata AS t1 
JOIN ravedr6.dr6_madera AS t2 ON t1.rave_obs_id = t2.rave_obs_id 
JOIN ravedr6.dr6_spectra AS t3 ON t1.rave_obs_id = t3.rave_obs_id 
AND t2.algo_conv_madera = 1
